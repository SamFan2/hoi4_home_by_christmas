add_namespace = hbc_japan

# Annexation of Korea
country_event = {
	id = hbc_japan.1
	title = hbc_japan.1.t
	desc = hbc_japan.1.d
	picture = GFX_hbc_event_japanese_troops_marching
	
	fire_only_once = yes
	
	is_triggered_only = yes

	option = { #Accept annexation
		name = hbc_japan.1.o1
		ai_chance = { 
			factor = 10 
			modifier = {
				is_historical_focus_on = yes
				factor = 100
			}
		}
		JAP = {
			country_event =  { hours = 1 id = hbc_japan.2 }
		}
		custom_effect_tooltip = GAME_OVER_TT
	}
	option = { #Refuse Annexation
		name = hbc_japan.1.o2
		ai_chance = { 
			factor = 1
			modifier = {
				is_historical_focus_on = yes
				factor = 0
			}
		}
		JAP = {
			country_event =  { hours = 1 id = hbc_japan.3 }
		}
	}
}
country_event = { #Korea accepts annexation
	id = hbc_japan.2
	title = hbc_japan.2.t
	desc = hbc_japan.2.d
	picture = GFX_hbc_event_japanese_troops_marching
	
	fire_only_once = yes
	
	is_triggered_only = yes

	option = { #Japan annexes korea
		name = hbc_japan.2.o1
		ai_chance = { 
			factor = 100
		}
		annex_country = { target = KOR transfer_troops = yes }
		hidden_effect = {
			news_event = { hours = 6 id = hbc_news.1 }
		}
	}
}
country_event = { #Korea refuses annexation
	id = hbc_japan.3
	title = hbc_japan.3.t
	desc = hbc_japan.3.d
	picture = GFX_hbc_event_japanese_troops_marching
	
	fire_only_once = yes
	
	is_triggered_only = yes

	option = { #Declare war on Korea
		name = hbc_japan.3.o1
		ai_chance = { 
			factor = 100
		}
		JAP = { end puppet = KOR }
		JAP = { declare_war_on = KOR
		hidden_effect = {
			news_event = { hours = 6 id = hbc_news.2 }
		}
	}
}
